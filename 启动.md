# 重启/停止

# 重新启动后端
~~~bash
cd /Users/zhangjinpeng/git/keep && poetry run env \
  PORT=8080 SECRET_MANAGER_TYPE=FILE SECRET_MANAGER_DIRECTORY=./state/ \
  DATABASE_CONNECTION_STRING='sqlite:///./state/db.sqlite3?check_same_thread=False' \
  PUSHER_APP_ID=1 PUSHER_HOST=localhost PUSHER_PORT=6001 \
  PUSHER_APP_KEY=keepappkey PUSHER_APP_SECRET=keepappsecret \
  USE_NGROK=false LOG_FORMAT=dev_terminal \
  python -m keep.cli.cli api --multi-tenant
  ~~~

# 重新启动前端
~~~bash
cd /Users/zhangjinpeng/git/keep/keep-ui && npm run dev
~~~
